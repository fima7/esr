\documentclass{beamer}

\usepackage{tikz}

\definecolor{structurecolor}{rgb}{0, .2, .5}
\definecolor{mybackground}{rgb}{1, .95, .8}
\definecolor{mybackground shadow}{rgb}{.7, .65, .5}
\setbeamercolor{background canvas}{fg=black,bg=mybackground}
\setbeamertemplate{navigation symbols}{}
\setbeamersize{text margin left=1em,text margin right=1em}

\begin{document}

\begin{frame}[t]
    \vspace{.2em}
    \begin{tikzpicture}[x=1em,y=1em,overlay]
      \draw[thick,color=structurecolor] (0, 0) -- ++ (\hsize, 0);
      \draw[thick,color=structurecolor] (0, 1em) -- ++ (\hsize, 0);
      \draw[thick,color=structurecolor] (0, -22.5em) -- ++ (\hsize, 0);

      \foreach \x in {1,2,...,31}
      {
        \draw[color=structurecolor](\x * \hsize/32, 0) -- ++ (0, -.1em);
        \draw[color=structurecolor](\x * \hsize/32, -22.5) -- ++ (0, -.1em);
      }
      \draw[color=structurecolor](\hsize/2, .1em) -- ++ (0, -.25em);
      \draw[color=structurecolor](\hsize/2, -22.4em) -- ++ (0, -.25em);

      % date & frame number
      \draw[thick,color=structurecolor] (\hsize-7.2em, 1.7em) -- (\hsize, 1.7em);

      % punch
      \foreach \y in {1.5,0.5,...,-22.5}
      {
        \shade[draw=none,left color=mybackground,right color=black] 
        (-.6, \y) circle (.2);
        \shade[color=white,inner color=gray!60] 
        (-.55, \y + 0.1) -- ++ (.05, -0.15) -- ++ (-.7, -.05) -- ++ (-.05, .15)
        -- cycle;
      }

      % shadow
      \draw[fill=black,draw=none] (-1, -22.85) rectangle ++ (.1, -.15);
      \draw[fill=mybackground shadow,draw=none] 
      (-.9, -22.85) rectangle ++ (34, -.15);

      \draw[fill=black,draw=none] (32.12, 1.96) rectangle ++ (.15, -.1);
      \draw[fill=mybackground shadow,draw=none] 
      (32.12, 1.86) rectangle ++ (.15, -25);
    \end{tikzpicture}
\end{frame}
\end{document}
